## 2. 系统架构设计 (System Architecture)

```mermaid
graph TD
    User[用户] --> View[Flet UI (View)]
    View --> Controller[Controller / Event Handlers]
    Controller --> Service[Service Layer (Business Logic)]
    Service --> Repository[Repository Layer]
    Repository --> DB[(SQLite Database)]
    Repository --> API[External AI API (Qwen)]
    
    subgraph "Python Runtime (Android)"
        View
        Controller
        Service
        Repository
        DB
    end
```

*   **View (UI)**：Flet 控件，只负责展示和事件捕获。
*   **Controller**：Flet 回调函数，调用 Service，更新 UI。
*   **Service**：业务逻辑核心（打卡计算、AI 解析、双账户校验）。
*   **Repository**：数据访问层，封装 SQLModel 操作。

## 3. 数据库设计 (Database Schema)

使用 **SQLModel** 定义，关键新增 **打卡记录表** 和 **分类表**。

### 3.1 任务表 (`Task`)

```python
from sqlmodel import SQLModel, Field
from datetime import datetime
from typing import Optional, List
from .checkin import CheckinRecord

class Task(SQLModel, table=True):
    __tablename__ = "tasks"
    
    id: Optional[int] = Field(default=None, primary_key=True)

    task_type: int = Field(default=0)                  # 0: Normal, 1: Reward
    current_streak: int = Field(default=0)             # 当前连续天数（冗余字段，可计算得出）
    max_streak: int = Field(default=0)                  # 历史最大连续
    last_completed_date: Optional[datetime] = Field(default=None)
    is_enabled: bool = Field(default=True)
    created_at: datetime = Field(default_factory=datetime.now)
    
    # 关联打卡记录（不存储在表中，用于 SQLModel 关系）
    # checkins: List["CheckinRecord"] = Relationship(back_populates="task")
```

### 3.2 打卡记录表 (`CheckinRecord`) **新增**

```python
class CheckinRecord(SQLModel, table=True):
    __tablename__ = "checkin_records"
    
    id: Optional[int] = Field(default=None, primary_key=True)
    task_id: int = Field(foreign_key="tasks.id", index=True)
    checkin_date: datetime                             # 打卡日期（仅日期部分有效）
    status: str = Field(default="normal")              # normal, overdue, missed
    actual_time: datetime = Field(default_factory=datetime.now)  # 实际打卡时间
```

**说明**：
- `checkin_date` 存储日期（00:00:00），用于判断连续。
- `status`：`normal` 正常打卡，`overdue` 补打卡，`missed` 自动标记未打卡。
- 连续天数可通过查询 `CheckinRecord` 按 `checkin_date` 排序计算，`Task` 表中 `current_streak` 作为冗余可加速显示，但每次打卡后需重新计算。

### 5.2 AI 服务调用

```python
import httpx
import json
from core.config import settings
from models.category import Category
from repositories.category_repo import get_all_categories

async def analyze_voice_text(text: str) -> dict:
    categories = await get_all_categories()
    category_names = [c.name for c in categories]
    
    prompt = f"""
    你是一个财务助手。从文本中提取：金额 (amount), 分类 (category), 是否奖金核销 (is_reimbursable), 置信度 (confidence 0-1)。
    文本：{text}
    可选分类：{category_names}
    返回严格 JSON 格式，例如：
    {{"amount": 35.5, "category": "餐饮", "is_reimbursable": true, "confidence": 0.95, "summary": "买咖啡"}}
    """
    
    url = f"{settings.api_base_url}/chat/completions"
    headers = {
        "Authorization": f"Bearer {settings.api_key}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": settings.model_name,
        "messages": [{"role": "user", "content": prompt}],
        "response_format": {"type": "json_object"}
    }
    
    async with httpx.AsyncClient() as client:
        try:
            resp = await client.post(url, json=payload, headers=headers, timeout=15.0)
            resp.raise_for_status()
            data = resp.json()
            content = data['choices'][0]['message']['content']
            return json.loads(content)
        except Exception as e:
            return {"error": str(e), "fallback": True}
```

### 5.3 补卡逻辑 (`services/streak_service.py`)

```python
from datetime import datetime, timedelta
from sqlmodel import select
from models.checkin import CheckinRecord
from repositories.database import get_session

async def get_streak(task_id: int) -> int:
    """计算当前连续天数（基于打卡记录）"""
    async with get_session() as session:
        # 获取所有打卡记录，按日期排序
        records = await session.exec(
            select(CheckinRecord)
            .where(CheckinRecord.task_id == task_id)
            .order_by(CheckinRecord.checkin_date.desc())
        )
        # 计算连续（从今天往前推，遇到 missed 或缺口停止）
        today = datetime.now().replace(hour=0, minute=0, second=0, microsecond=0)
        streak = 0
        for i, rec in enumerate(records):
            expected_date = today - timedelta(days=i)
            if rec.checkin_date.date() == expected_date.date() and rec.status != "missed":
                streak += 1
            else:
                break
        return streak

async def补卡(task_id: int, target_date: datetime):
    """补打卡，标记为 overdue"""
    async with get_session() as session:
        record = CheckinRecord(
            task_id=task_id,
            checkin_date=target_date.replace(hour=0, minute=0, second=0, microsecond=0),
            status="overdue",
            actual_time=datetime.now()
        )
        session.add(record)
        await session.commit()
        # 更新任务表中的冗余 streak（可选）
        # 也可以不更新，显示时实时计算
```

### 5.4 备份与导入 (`services/backup_service.py`)

使用 Flet 的 `FilePicker`：

```python
import flet as ft
import shutil
from pathlib import Path

async def export_db(page: ft.Page):
    # 复制数据库文件到用户可访问位置
    db_path = Path("data.db")  # 实际路径
    backup_path = await page.pick_file(save=True)  # 用户选择保存位置
    if backup_path:
        shutil.copy(db_path, backup_path)
        page.show_snack_bar(ft.SnackBar(content=ft.Text("备份成功")))

async def import_db(page: ft.Page):
    # 选择备份文件
    file = await page.pick_file()
    if file:
        # 自动备份当前数据库
        db_path = Path("data.db")
        backup_current = Path("data_backup_before_import.db")
        shutil.copy(db_path, backup_current)
        # 覆盖
        shutil.copy(file.path, db_path)
        # 提示重启
        page.dialog = ft.AlertDialog(
            title=ft.Text("导入完成"),
            content=ft.Text("请重启应用以生效"),
            actions=[ft.TextButton("确定", on_click=lambda e: page.window.close())]
        )
        page.dialog.open = True
        page.update()
```

## 8. 风险评估 (更新版)

| 风险点 | 可能性 | 影响 | 应对策略 |
